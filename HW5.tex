\section*{Problem 1}
Find closed-form expressions for the following sums/products.
\begin{enumerate}[label=\arabic*.]
    \item $\displaystyle \sum_{i=1}^n (2i + 1)$
    \item $\displaystyle \sum_{i=1}^n \sum_{j=1}^m (i + j^2)$
    \item $\displaystyle \sum_{i=1}^n \sum_{j=1}^m 2^{i+2j}$
    \item $\displaystyle \prod_{i=1}^n \prod_{j=1}^m 2^i 3^j$
\end{enumerate}
\medskip
\solution

\begin{enumerate}[label=\arabic*.]
    \item 

    \[
        \sum_{i=1}^n (2i+1) = \sum_{i=1}^n 2i + \sum_{i=1}^n 1 = 2\sum_{i=1}^n i + \sum_{i=1}^n 1 = 2\left(\frac{n(n+1)}{2}\right) + n
    \]

    \[
        \boxed{2\left(\frac{n(n+1)}{2}\right) + n}
    \]

    \medskip
    \begin{center}
    \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip

    \item

    \begin{align*}
        \sum_{i=1}^n \sum_{j=1}^m (i + j^2) &= \sum_{i=1}^n \left((i+1^2) + (i+2^2) + \cdots + (i+m^2)\right) \\
        &= \sum_{i=1}^n \left(im + (1^2 + 2^2 + 3^2 + \cdots + m^2)\right) \\
        &= 1m(1^2 + 2^2 + 3^2 + \cdots + m^2) + 2m(1^2 + 2^2 + 3^2 + \cdots + m^2) + ... + nm(1^2 + 2^2 + 3^2 + \cdots + m^2) \\
        &= (1^2 + 2^2 + \cdots + m^2)\left(1m + 2m + \cdots + nm\right) \\
        &= \left(\frac{m(m+1)(2m+1)}{6}\right) \cdot m \cdot \left(\frac{n(n+1)}{2}\right)
    \end{align*}

    \[
        \boxed{\left(\frac{m(m+1)(2m+1)}{6}\right) \cdot m \cdot \left(\frac{n(n+1)}{2}\right)}
    \]

    \medskip
    \begin{center}
    \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip

    \item

    \begin{align*}
        \sum_{i=1}^n \sum_{j=1}^m 2^{i+2j} &= \sum_{i=1}^n \left(2^{i+2(1)} + 2^{i+2(2)} + 2^{i+2(3)} + \cdots + 2^{i+2m}\right) \\
        &= \sum_{i=1}^n \left(2^i \cdot 2^{2} + 2^i \cdot 2^{4} + 2^i \cdot 2^{6} + \cdots + 2^i \cdot 2^{2m}\right) \\
        &= \sum_{i=1}^n 2^i \left(2^{2(1)} + 2^{2(2)} + 2^{2(3)} + \cdots + 2^{2m}\right)
    \end{align*}

    let 

    \[S = 2^{2(1)} + 2^{2(2)} + 2^{2(3)} + \cdots + 2^{2m}\]

    , and 

    \[
        2^2 S = 2^2 + 2 + 2^{2(2)} + 2 + 2^{2(3)} + 2 + \cdots + 2^{2m} + 2
    \]

    subtract the two

    \[
        2^2 S - S = 2^{2m+2} - 2^{2(1)} \Rightarrow S = \frac{2^{2m+2} - 2^2}{2^2 - 1}
    \]

    then, we continue

    \[
        \sum_{i=1}^n 2^i \left(\frac{2^{2m+2} - 4}{3}\right) = 2^1\left(\frac{2^{2m+2} - 4}{3}\right) + 2^2\left(\frac{2^{2m+2} - 4}{3}\right) + \cdots + 2^n\left(\frac{2^{2m+2} - 4}{3}\right)
    \]

    \[
        = \left(\frac{2^{2m+2} - 4}{3}\right)\left(2^1 + 2^2 + \cdots + 2^n\right)
    \]

    let $S' = 2^1 + 2^2 + 2^3 + \cdots + 2^n$, and $2S' = 2^2 + 2^3 + 2^4 + \cdots + 2^{n+1}$, we find 

    \[
        2S' - S' = 2^{n+1} - 2^1 \Rightarrow S' = 2^{n+1} - 2
    \]

    and therefore

    \[
        \boxed{\left(\frac{2^{2m+2} - 4}{3}\right)(2^{n+1} - 2)}
    \]

    \medskip
    \begin{center}
    \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip

    \item 

    \begin{align*}
        \prod_{i=1}^n \prod_{j=1}^m 2^i \cdot 3^j &= \prod_{i=1}^n \left((2^i \cdot 3^1)(2^i \cdot 3^2)(2^i \cdot 3^3) \cdots (2^i \cdot 3^m)\right) \\
        &= \prod_{i=1}^n \left(2^{im} \cdot 3^{\frac{m(m+1)}{2}}\right) \\
        &= \left(2^m \cdot 3^{\frac{m(m+1)}{2}} \cdot 2^{2m} \cdot 3^{\frac{m(m+1)}{2}} \cdot ... \cdot 2^{nm} \cdot 3^{\frac{m(m+1)}{2}}\right) \\
        &= \left(3^{\frac{m(m+1)}{2}} \cdot 2^{(m + 2m + 3m + ... + nm)}\right)
    \end{align*}

    \[
        \boxed{\left(3^{\frac{m(m+1)}{2}}\right)^n \cdot 2^{m\left(\frac{n(n+1)}{2}\right)}}
    \]
\end{enumerate}

\section*{Problem 2}
Use integral bounds to give upper and lower bounds for
\begin{enumerate}[label=\arabic*.]
    \item $\displaystyle \sum_{x=1}^n \frac{1}{x^2}$
    \item $\displaystyle \sum_{x=1}^n x^{3/2}$
\end{enumerate}
\medskip
\solution

\begin{enumerate}[label=\arabic*.]
    \item $\displaystyle \sum_{x=1}^n \frac{1}{x^2}$ we let $f(x)=\frac{1}{x^2}$, this is a decreasing function so

    \[
        \int_{1}^{n+1} f(x)\,dx \le \sum_{k=1}^n f(k) \le \int_{1}^{n} f(x)\,dx + f(1)
    \]

    find the integral first

    \begin{align*}
    \int f(x)\,dx &= \int \frac{1}{x^2}\,dx = \int x^{-2}\,dx \\
        &= x^-1 + c
    \end{align*}
    then we can calculate the bound

    \[
        \left(-\frac{1}{x}\right)\Bigg\rvert_{1}^{n+1} \le \sum_{k=1}^n f(k) \le \left(-\frac{1}{x}\right)\Bigg\rvert_{1}^{n} + f(1)
    \]

    \[
        -\frac{1}{n+1} - (-1) \le \sum_{k=1}^n f(k) \le -\frac{1}{n} - (-1) + 1
    \]

    \[
        \boxed{1 - \frac{1}{n+1} \le \sum_{x=1}^n \frac{1}{x^2} \le 2 - \frac{1}{n}}
    \]

\section*{Problem 4: Data Structure Cheat Sheet}
Solve or verify the following recurrences (assume $T(1)=1$).
\begin{enumerate}[label=\arabic*.]
    \item $T(n) = T(n-1) + n$ (verify by induction).
    \item $T(n) = T(n-1) + n^3$.
    \item $T(n) = T(\lfloor n/2 \rfloor) + 1$.
    \item $T(n) = T(\lfloor n/2 \rfloor) + n$.
    \item $T(n) = 2T(\lfloor n/2 \rfloor) + n$.
    \item $T(n) = 2T(\lfloor n/2 \rfloor) + 1$.
    \item (Optional) $T(n) = 2T(\lfloor n/2 \rfloor) + \log_2 n$.
\end{enumerate}
\medskip
\solution

\begin{enumerate}[label=\arabic*.]
    \item $T(n)=T(n-1)+n$ ; $T(1)=1$

    we expand the recurrence to observe the accumulating pattern, then summarize the resulting series to express the closed-form solution.

    \[
    \begin{aligned}
        T(1) &= 1 \\
        T(2) &= T(1) + 2 = 1 + 2 = 3 \\
        T(3) &= T(2) + 3 = 1 + 2 + 3 = 6 \\
        T(n) &= T(n-1) + n = 1 + 2 + 3 + \cdots + (n-1) + n \\
             &= \sum_{i=1}^n i \\
        $\therefore T(n)&=\dfrac{n(n+1)}{2}$
    \end{aligned}
    \]

    now we prove $T(n)=\dfrac{n(n+1)}{2}$ with induction

    \vspace{1em}

    the claim 

    \[P(n)=T(n)=\dfrac{n(n+1)}{2}\]

    for the base case where $n_0=1$

    \[
    \begin{aligned}
        P(1) &= T(1) = \frac{1(1+1)}{2} = 1 \\
        P(2) &= T(2) = \frac{2(3)}{2} = 3 \\
        P(3) &= T(3) = \frac{3(4)}{2} = 6
    \end{aligned}
    \]

    we assume $P(k)$ for arbitrary $k \ge n_0$, then try to prove $P(k+1) \equiv (k+1)(k+2)/2$

    \[
    \begin{aligned}
        P(k+1) &= P(k) + (k+1) \\
               &= \frac{k(k+1)}{2} + \frac{2(k+1)}{2} \quad \text{by Inductive Hypothesis} \\
               &= \frac{(k+1)(k+2)}{2}
    \end{aligned}
    \]

    since $P(k+1)$ comes after $P(k)$, our induction case is complete, and we can conclude the claim is true.

    \[
        \boxed{T(n)=\frac{n(n+1)}{2}}
    \]

    \medskip
    \begin{center}
        \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip

    \item $T(n)=T(n-1)+n^3$ ; $T(1)=1$

    we expand the recurrence to observe the accumulating pattern, then summarize the resulting series to express the closed-form solution.

    \[
    \begin{aligned}
        T(1) &= 1 \\
        T(2) &= T(1) + 2^3 = 1 + 2^3 \\
        T(3) &= T(2) + 3^3 = (1 + 2^3) + 3^3 \\
        \vdots \\
        T(n) &= T(n-1) + n^3 = \sum_{i=1}^n i^3 \\
        \therefore T(n)&=\left(\dfrac{n(n+1)}{2}\right)^2 && \quad \text{by the sum of cube formula}
    \end{aligned}
    \]


    \[
        \boxed{T(n)=\left(\frac{n(n+1)}{2}\right)^2}
    \]

    \medskip
    \begin{center}
        \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip

    \item $T(n)=T\left(\frac{n}{2}\right)+1$ ; $T(1)=1$

    we expand the recurrence to observe the accumulating pattern, then summarize the resulting series to express the closed-form solution.

    \[
    \begin{aligned}
        T(2^0) &= 1 = T(2^0) \\
        T(2^1) &= T(1) + 1 = 1 + 1 = 2 = T(2^1) \\
        T(2^2) &= T(2^1) + 1 = 2 + 1 = 3 \\
        T(2^3) &= T(2^2) + 1 = 3 + 1 = 4 \\
        \vdots \\
        T(2^n^') &= n + 1
    \end{aligned}
    \]

    let $n = {2^{n^'}} \rightarrow {n^'} = \log_2 n \text{ which gives }$

    \[
        \boxed{T(n)=\log_2 n + 1}
    \]

    \medskip
    \begin{center}
        \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip

    \item $T(n)=T\left(\frac{n}{2}\right)+n$ ; $T(1)=1$

    we expand the recurrence to observe the accumulating pattern, then summarize the resulting series to express the closed-form solution.

    \[
    \begin{aligned}
        T(2^0) &= T(1) = 1 \\
        T(2^1) &= T(2^0) + 2^1 = 1 + 2 = 3 \\
        T(2^2) &= T(2^1) + 2^2 = 3 + 4 = 7 \\
        \vdots \\
        T(2^n) &= T(2^{n-1}) + 2^n = \sum_{i=0}^n 2^i
    \end{aligned}
    \]

    we let 

    \[S = 2^0 + 2^1 + 2^2 + \cdots + 2^n \tag{i}\]

    then
    \begin{align*}
    2S &= 2^1 + 2^2 + 2^3 + \cdots + 2^{n+1} \tag{ii} \\
    (ii) - (i) : S &= 2^{n+1} - 1 \\
    \end{align*}

    therefore $T(2^n) &= 2^{n+1} - 1$, let $n' = 2^n$  &\rightarrow n = \log_2 n'$ which gives$

    \[
        \boxed{T(n)=2^{\log_2 n + 1} - 1}
    \]

    \medskip
    \begin{center}
        \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip

    \item $T(n)=2T\left(\frac{n}{2}\right)+n$ ; $T(1)=1$

    we expand the recurrence to observe the accumulating pattern, then summarize the resulting series to express the closed-form solution.

    \[
    \begin{aligned}
        T(2^0) &= 2^0 = 1 \\
        T(2^1) &= 2T(2^0) + 2^1 = 2 \cdot 2^0 + 2^1 = 2^1 + 2^1 = 2(2^1) \\
        T(2^2) &= 2T(2^1) + 2^2 = 2(2^1 + 2^1) + 2^2 = 2^2 + 2^2 + 2^2 = 3(2^2) \\
        T(2^3) &= 2T(2^2) + 2^3 = 2(2^2 + 2^2 + 2^2) + 2^3 = 2^3 + 2^3 + 2^3 + 2^3 = 4(2^3) \\
        \vdots \\
        T(2^n) &= (n+1)(2^n)
    \end{aligned}
    \]

    let $n' = 2^n \rightarrow n = \log_2 n'$ which gives

    \[
        \boxed{T(n) = (\log_2 n + 1)\left(2^{\log_2 n}\right)}
    \]

    \medskip
    \begin{center}
        \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip

    \item $T(n)=2T\left(\frac{n}{2}\right)+1$ ; $T(1)=1$

    we expand the recurrence to observe the accumulating pattern, then summarize the resulting series to express the closed-form solution.

    \[
    \begin{aligned}
        T(2^0) &= 2^0 \cdot T(2^0) \\
        T(2^1) &= 2T(2^0) + 1 = 2 \cdot 2^0 + 1 \\
        T(2^2) &= 2T(2^1) + 1 = 2(2^0 + 1) + 1 = 2^2 + 2 + 1 \\
        T(2^3) &= 2T(2^2) + 1 = 2(2^2 + 2 + 1) + 1 = 2^3 + 2^2 + 2 + 1 \\
        \vdots \\
        T(2^n) &= \sum_{i=0}^n 2^i = 2^{n+1} - 1 \quad (\text{already prove in 4.4})
    \end{aligned}
    \]

    let $n' = 2^n \rightarrow n = \log_2 n'$ which gives

    \[
        \boxed{T(n)=2^{\log_2 n + 1} - 1}
    \]
\end{enumerate}

\section*{Problem 5}
For each pair $(f,g)$, determine which of $\{\sim, o, O, \omega, \Omega, \Theta\}$ apply.
\begin{enumerate}[label=\arabic*.]
    \item $f(n) = 30n + 900\log n$, $g(n) = n$
    \item $f(n) = \log n$, $g(n) = n$
    \item $f(n) = (\log n)^2$, $g(n) = n$
    \item $f(n) = \sqrt{n}$, $g(n) = (\log n)^{999}$
    \item $f(n) = n2^n$, $g(n) = n$
    \item $f(n) = n^2$, $g(n) = 1.0000000001^n$
    \item $f(n) = 200{,}000$, $g(n) = 1$
    \item $f(n) = 2^n$, $g(n) = 10^n$
\end{enumerate}
\medskip
\solution

\begin{enumerate}[label=\arabic*.]
    \item $f(n)=30n+900\log n$ , $g(n)=n$

    we evaluate the limit of $\dfrac{f(n)}{g(n)}$ to determine all applicable relationships.

    \[
        \lim_{x \to \infty} \frac{f(n)}{g(n)} = \lim_{x \to \infty} \frac{30n + 900 \log n}{n} = \lim_{x \to \infty} \frac{30 + 900 \left(\frac{1}{n \ln 2}\right)}{1} = 30
    \]

    \[
        \boxed{f \in O(g),\ f \in \Omega(g)}
    \]

    \medskip
    \begin{center}
        \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip

    \item $f(n)=\log n$ , $g(n)=n$

    we evaluate the limit of $\dfrac{f(n)}{g(n)}$ to determine all applicable relationships.

    \[
        \lim_{x \to \infty} \frac{\log n}{n} = 0
    \]

    \[
        \boxed{f \in o(g),\ f \in O(g)}
    \]

    \medskip
    \begin{center}
        \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip

    \item $f(n)=\log_2 n$ , $g(n)=n$

    we evaluate the limit of $\dfrac{f(n)}{g(n)}$ to determine all applicable relationships.

    \[
        \lim_{x \to \infty} \frac{\log_2 n}{n} = \lim_{x \to \infty} \frac{1}{n \ln 2} = 0
    \]

    \[
        \boxed{f \in O(g),\ f \in o(g)}
    \]

    \medskip
    \begin{center}
        \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip

    \item $f(n)=n$ , $g(n)=(\log n)^{999}$

    we evaluate the limit of $\dfrac{f(n)}{g(n)}$ to determine all applicable relationships.

    \[
    \begin{aligned}
        \lim_{x \to \infty} \frac{(n)^{1/2}}{(\log n)^{999}} &= \lim_{x \to \infty} \frac{\tfrac{1}{2} n^{-1/2}}{999 \cdot (\log n)^{998} \cdot \tfrac{1}{n \ln 2}} \\
        &= \lim_{x \to \infty} \frac{n \ln 2}{\tfrac{1}{2} \sqrt{n} \cdot 999 \cdot (\log n)^{998}}  \\
        &= \lim_{x \to \infty} \frac{\ln 2}{\text{\{some term  with n\}}} = 0
    \end{aligned}
    \]

    \[
        \boxed{f \in O(g),\ f \in o(g)}
    \]

    \medskip
    \begin{center}
        \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip

    \item $f(n)=n2^n$ , $g(n)=n$

    we evaluate the limit of $\dfrac{f(n)}{g(n)}$ to determine all applicable relationships.

    \[
        \lim_{n \to \infty} \frac{n2^n}{n} = 2^n = \infty
    \]

    \[
        \boxed{f \in \omega(g),\ f \in \Omega(g)}
    \]

    \medskip
    \begin{center}
        \rule{0.3\linewidth}{0.2pt}
    \end{center}

\section*{Problem 6}
Solve the recurrences:
\begin{enumerate}[label=\arabic*.]
    \item $T_{i+1} = 5T_i - 6T_{i-1}$ with $T_0 = 8$, $T_1 = 17$
    \item $T_{i+1} = -T_i + 12T_{i-1}$ with $T_0 = 0$, $T_1 = -7$
\end{enumerate}

\medskip
\solution

We solve each recurrence by writing $T_i = x^i$, finding the characteristic equation, and then applying the initial conditions to obtain the constants.

\begin{enumerate}
    \item we'd want to write the $i^{th}$ term of $T_i$ in a form of $x^i$, so we let

        \[T_i = x^i\]

        thus 

        \[T_{i+1} = 5T_i - 6T_{i-1} \equiv x^{i+1} = 5x^i - 6x^{i-1}\]

        divide the newly created equation with x^{i-1} gives

        \[x^2 - 5x + 6 = 0\]

        and we solve this characteristic equation with quadratic formula

        \begin{align*}
            x &= \frac{-(-5) \pm \sqrt{(-5)^2 - 4(1)(6))}}{2(1)} \\
            x &= \frac{5 + 1}{2}, \frac{5 - 1}{2} \\
            x &= 2, 3
        \end{align*}

        the solution could be formed from $2^i$ and $3^i$  by adjusting the two constant attached to them

        \[T_i = c_1 2^i + c_2 3^i\]

        we find those two constant from the given $T_0$ and $T_1$ as

        \begin{align*}
        T_0 &= 8 = c_1 2^0 + c_2 3^0 \\
            &= 8 = c_1 + c_2 \tag{i} 
        \end{align*}

        and 

        \begin{align*}
        T_1 &= 17 = c_1 2^1 + c_2 3^1 \\
            &= 17 = 2c_1 + 3c_2 \tag{ii}
        \end{align*}

        solving this linear system gives $c_1 = 7 \text{, } c_2 = 1$, and our closed form solution is complete

        \[\boxed{T_{i+1} = 7 \cdot 2^{i+1} + 3^{i+1}}\]

        \[\cong \boxed{T_{i} = 7 \cdot 2^{i} + 3^{i}}\]

    \medskip
    \begin{center}
    \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip    

    \item we'd want to write the $i^{th}$ term of $T_i$ in a form of $x^i$, so we let

        \[T_i = x^i\]

        thus 

        \[T_{i+1} = -T_i + 12T_{i-1} \equiv x^{i+1} = -x^i + 12x^{i-1}\]

        divide the newly created equation with x^{i-1} gives

        \[x^2 + x - 12 = 0\]

        and we solve this characteristic equation with quadratic formula

        \begin{align*}
            x &= \frac{-1 \pm \sqrt{(1)^2 - 4(1)(-12)}}{2(1)} \\
            x &= \frac{-1 + 7}{2}, \frac{-1 - 7}{2} \\
            x &= 3, -4
        \end{align*}

        the solution could be formed from $3^i$ and $(-4)^i$  by adjusting the two constant attached to them

        \[T_i = c_1 3^i + c_2 (-4)^i\]

        we find those two constant from the given $T_0$ and $T_1$ as

        \begin{align*}
        T_0 &= 0 = c_1 3^0 + c_2 (-4)^0 \\
            &= 0 = c_1 + c_2 \tag{i} 
        \end{align*}

        and 

        \begin{align*}
        T_1 &= -7 = c_1 3^1 + c_2 (-4)^1 \\
            &= -7 = 3c_1 - 4c_2 \tag{ii}
        \end{align*}

        solving this linear system gives $c_1 = -1 \text{, } c_2 = 1$, and our closed form solution is complete

        \[\boxed{T_{i+1} = -1 \cdot 3^{i+1} + (-4)^{i+1}}\]

        \[\cong \boxed{T_{i} = -1 \cdot 3^{i} + (-4)^{i}}\]
\end{enumerate}
