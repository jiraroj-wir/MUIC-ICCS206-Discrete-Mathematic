\section*{Problem 1: More Counting}
\begin{enumerate}[label=\arabic*.]
    \item Let $x_1,\dots,x_k \in \{1,2,3,\dots\}$. How many solutions satisfy $\sum_{i=1}^k x_i = n$?
    \item Same domain as above. How many solutions satisfy $\sum_{i=1}^k x_i \le n$?
    \item Let $x_1,\dots,x_k \in \{0,1,2,3,\dots\}$. How many solutions satisfy $\sum_{i=1}^k x_i = n$?
    \item Same domain as above. How many solutions satisfy $\sum_{i=1}^k x_i \le n$?
\end{enumerate}

\medskip
\solution

\begin{enumerate}
    \item we try to frame this as star-and-bars, and as 0 (empty) is allowed, we have to shift the set of $\{1, 2, 3, ...\}$ down by 1, as ${{x^'}_i} = x_i - 1$, so the set would be $\{0, 1, 2, ...\}$ instead; and the summation becomes 
        \[\sum_{i=1}^k {{x^'}_i} = n - k\]
        now, to find all the permutation that satisfy the given criteria, we can apply star-and-bars --- $(n-k)$ stars, k bars --- which gives $\dbinom{(n - k) + k - 1}{k-1} = \boxed{\dbinom{n-1}{k-1}}$ solutions
    \item just like (1.1), we try to frame it as star-and-bars by introduces 0 (emptyness) into the set, but for this one, we have to accounted for the inequality by assigning anouther spot `s` such that $0 \le s \le (n-k)$, and the summation becomes
        \[\sum_{i=1}^k {{x^'}_i} + s= n - k\]
        the star counts remain the same, but as we add another spot/ box, there are now k+1 bars --- which gives $\dbinom{(n - k) + k + 1 - 1}{k + 1 - 1}= \boxed{\dbinom{n}{k}}$ solutions.
    \item this one is just (1.1), but the set already in a proper condition for star-and-bars; so from the summation, just plug it in --- n stars, k bars --- $\boxed{\dbinom{n+k-1}{k-1}}$ solutions.
    \item like (1.2), followed from (1.3), we could rewrite the summation as
        \[\sum_{i=1}^k x_i + s = n \quad \text{ where } 0 \le s \le n\]
        the star counts remain the same, but as we add another spot/ box, there are now k+1 bars --- which gives $\boxed{\dbinom{n+k}{k}}$
\end{enumerate}

\section*{Problem 2}
From 100 classes you record whether AJ Piti ends late and whether it rains:
\[
\begin{array}{c|cc}
& \text{Late} & \text{Not Late} \\
\hline
\text{Rain} & 10 & 15 \\
\text{Not Rain} & 50 & 25
\end{array}
\]
Compute:
\begin{enumerate}[label=\arabic*.]
    \item $P(\text{Rain})$
    \item $P(\text{Late})$
    \item $P(\text{Late} \mid \text{Rain})$
    \item $P(\text{Late} \cap \text{Rain})$
    \item $P(\text{Rain} \mid \text{On Time})$
    \item Are ``Late'' and ``Rain'' independent?
\end{enumerate}

\medskip
\solution

\begin{enumerate}
    \item $\dfrac{25}{100} = \dfrac{1}{4}$
    \item $\dfrac{60}{100} = \dfrac{3}{5}$
    \item $\dfrac{P(\text{Late})}{P(\text{Rain})} = \dfrac{10}{25} = \dfrac{2}{5}$
    \item $\dfrac{10}{100} = \dfrac{1}{10}$
    \item $\dfrac{P(\text{Rain})}{P(\text{Not Late})} = \dfrac{15}{40} = \dfrac{3}{8}$
    \item no, as \[P(\text{Late $\cap$ Rain}) = \dfrac{1}{10} \ne \left(\dfrac{3}{5}\right)\left(\dfrac{1}{4}\right) = \dfrac{3}{20} = P(\text{Rain}) \cdot P(\text{Late})\]
\end{enumerate}

\section*{Problem 3}
A bag initially contains 2 cola and 3 grape yoyos.
\begin{enumerate}[label=\arabic*.]
    \item Draw two yoyos sequentially without replacement.
    \begin{enumerate}[label=\alph*.]
        \item Probability of one cola and one grape.
        \item Probability of two colas.
        \item Probability the first yoyo is cola.
        \item Given the second yoyo is grape, probability the first is cola.
        \item Given the first is cola, probability the two yoyos are different flavors.
        \item Given the two yoyos are different flavors, probability the first is cola.
        \item Given the two yoyos are the same flavor, probability the first is cola.
    \end{enumerate}
    \item Draw two yoyos simultaneously (without replacement).
    \begin{enumerate}[label=\alph*.]
        \item Probability of one cola and one grape.
        \item Probability of two colas.
    \end{enumerate}
    \item Draw the first yoyo, replace it, then draw a second.
    \begin{enumerate}[label=\alph*.]
        \item Probability of one cola and one grape.
        \item Probability of two colas.
        \item Probability the first yoyo is cola.
        \item Given the second is grape, probability the first is cola.
        \item Given the first is cola, probability the two yoyos are different flavors.
        \item Given the two are different flavors, probability the first is cola.
        \item Given the two are the same flavor, probability the first is cola.
    \end{enumerate}
\end{enumerate}

\medskip
\solution

\begin{enumerate}
    \item let $C$ notates cola, and $G$ notates grape
    \begin{enumerate}
        \item $\dfrac{2}{5}\cdot\dfrac{3}{4} + \dfrac{3}{5}\cdot\dfrac{2}{4} = \dfrac{12}{20} = \boxed{\dfrac{3}{5}}$
        \item $\dfrac{2}{5}\cdot\dfrac{1}{4} = \boxed{\dfrac{1}{10}}$
        \item $\boxed{\dfrac{2}{5}}$
        \item $\dfrac{2}{4} = \boxed{\dfrac{1}{2}}$
        \item there'll be 1 cola, 3 grapes left, so the probability that we get grape is $\boxed{\dfrac{3}{4}}$
        \item $P(CG) = P(GC)$, so $\boxed{\dfrac{1}{2}}$
        \item if the first yoyo gonna be a cola, it must be $CC$, so
            \[\dfrac{P(CC)}{P(CC)+P(GG)} = \dfrac{\frac{1}{10}}{\frac{1}{10}+\frac{3}{10}} = \boxed{\dfrac{1}{4}}\]
    \end{enumerate}

    \medskip
    \begin{center}
    \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip

    \item
    \begin{enumerate}
        \item $\dfrac{\binom{2}{1}\binom{3}{1}}{\binom{5}{2}} = \boxed{\dfrac{3}{5}}$
        \item $\dfrac{\binom{2}{2}}{\binom{5}{2}} = \boxed{\dfrac{1}{10}}$
    \end{enumerate}

    \medskip
    \begin{center}
    \rule{0.3\linewidth}{0.2pt}
    \end{center}
    \medskip

    \item 
    \begin{enumerate}
        \item $2\left(\dfrac{2}{5}\cdot\dfrac{3}{5} \right)= \boxed{\dfrac{12}{25}}$
        \item $\left(\dfrac{2}{5}\right)^2 = \boxed{\dfrac{4}{25}}$
        \item $\boxed{\dfrac{2}{5}}$
        \item the first yoyo got put back anyway, so $\boxed{\dfrac{2}{5}}$
        \item it's just the probability that we got grape for second draw (after put back the first yoyo), so $\boxed{\dfrac{3}{5}}$
        \item $\boxed{\dfrac{1}{2}}$
        \item $\dfrac{P(CC)}{P(CC)+P(GG)} = \dfrac{\frac{4}{25}}{\frac{4}{25}+\frac{9}{25}} = \boxed{\dfrac{4}{13}}$
    \end{enumerate}
\end{enumerate}

\section*{Problem 5}
Three identical boxes each contain two yoyos. Two boxes have one cola and one grape; the third has two colas. Play the game:
\begin{enumerate}[label=\arabic*.]
    \item Pick a box.
    \item Draw one yoyo.
    \item If it's grape, return it and reselect a box; repeat.
    \item If it's cola, keep drawing from that box. If the next yoyo is cola AJ wins; otherwise you win.
\end{enumerate}
Is this a fair game? Justify.

\section*{Problem 6}
AJ Piti will give a rare banana yoyo to Bossy, Tow$+$, or John, each equally likely. AJ secretly selects the recipient. If Bossy asks for one name who will not get the yoyo, AJ responds as follows:
\begin{itemize}
    \item If Bossy does not get the yoyo, AJ names the other person who also does not get it.
    \item If Bossy does get it, AJ flips a fair coin to name John or Tow$+$.
\end{itemize}
AJ tells Bossy that John will not get the yoyo. Bossy concludes his chance is $1/2$, while Tow$+$ claims it remains $1/3$ for Bossy and $2/3$ for himself. Determine the actual probabilities for Bossy and Tow$+$ given AJ's statement. (Hint: draw a probability tree.)
